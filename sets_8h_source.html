489 inline RCP<const Set> imageset(const RCP<const Basic> &sym,
490  const RCP<const Basic> &expr,
491  const RCP<const Set> &base)
492 {
493  if (not is_a_sub<Symbol>(*sym))
494  throw SymEngineException("first arg is expected to be a symbol");
495 
496  if (eq(*expr, *sym) or eq(*base, *emptyset()))
497  return base;
498 
499  if (is_a_Number(*expr))
500  return finiteset({expr});
501  if (is_a_Set(*expr)) {
502  for (const auto &s : static_cast<const Set &>(*expr).get_args()) {
503  if (not(is_a_Number(*s) or is_a<Constant>(*s)
504  or is_a_Boolean(*s))) {
505  return make_rcp<const ImageSet>(sym, expr, base);
506  }
507  }
508  return finiteset({expr});
509  }
510 
511  if (is_a<FiniteSet>(*base)) {
512  map_basic_basic d;
513  set_basic temp;
514  for (const auto &s :
515  down_cast<const FiniteSet &>(*base).get_container()) {
516  d[sym] = s;
517  temp.insert(expr->subs(d));
518  d.clear();
519  }
520  return finiteset(temp);
521  }
522 
523  if (is_a<ImageSet>(*base)) {
524  const ImageSet &imbase = down_cast<const ImageSet &>(*base);
525  map_basic_basic d;
526  d[sym] = imbase.get_expr();
527  return imageset(imbase.get_symbol(), expand(expr->subs(d)),
528  imbase.get_baseset());
529  }
530 
531  return make_rcp<const ImageSet>(sym, expr, base);
532 }
