70 
71  if (eq(**imag_, *zero)) {

73  *imag_ = zero;
74  return;
75  }
76  if (is_a<Integer>(*exp_)) {
77  RCP<const Basic> expx;
78  if (eq(*Gt(exp_, zero), *boolTrue)) {
79  expx = expand(x.rcp_from_this());
80  } else {
81  auto magn = add(mul(*real_, *real_), mul(*imag_, *imag_));
82  *real_ = div(*real_, magn);
83  *imag_ = div(neg(*imag_), magn);
84  expx = expand(pow(add(*real_, mul(*imag_, I)), neg(exp_)));
85  }
86  if (eq(*expx, x))
87  throw SymEngineException("Not Implemented");
88  apply(*expx);
